## 1. 架构约定（必须沿用 · 最小改动）

库内已有**三个并列 per-module runner**（`skill-/tool-/memory-evaluation-runner.ts`），**不继承统一基类**，各自实现 + 共享 helper。新 lab 沿用——**不引入泛型 EvaluationRunner 基类**（避免接缝层扩张）。

每个 lab = 复用共享件 + 1 新 runner + 1 新 Pane：
- **共享后端**：`evaluation-common.ts`(`runJudge`)、`evaluation-errors.ts`、`evaluation-archive.ts`。
- **共享前端**：`web/src/lib/evaluation-export.ts`、`EvaluationArchiveIndexItem` 列表、`StatusIcon`/`inputClass`/`EmptyState`（从 `ToolLabPane` 对齐复制，不强抽公共组件库——与现状一致）。
- **请求解析**：仿 `skill-evaluation-api.ts` / `tool-evaluation-api.ts`。

**两种测评范式**（按资源性质二选一）：
- **A·LLM 运行评判**（prompts / subagents / command 实跑）→ 复刻 **skill-evaluation-runner**：`variants × tasks × repeat → 运行 → runJudge / pairwise`。
- **B·确定性断言**（hooks / command 展开）→ 复刻 **tool-evaluation-runner**：`cases × repeat → 确定执行 → 断言`。

## 2. 行业趋势 → 「完整测评」的内涵（agent harness 方向）

把 agent harness 行业趋势映射为 DoD，避免只做「能跑一遍」：

1. **Eval-as-first-class / LLM-judge + pairwise(Elo)**：prompts lab 直接继承 skill 的 win/tie/loss，作 A/B 核心读数；其余 lab 至少 rubric 打分 + `judgeRepeat` 降方差。
2. **轨迹/过程评测（不止终态）**：**subagents lab 必做**——捕获 `tool_call` 序列、步数、token 经济性，新增 `tool-sequence` / `step-budget` / `token-budget` 断言。
3. **护栏 policy-as-code 单测化**：**hooks lab = 护栏单测**——合成生命周期事件 fixture 喂 `px-hook-runner` 规则，断言 verdict(`must-allow`/`must-block`/`golden-mutation`/`trigger-count`)。
4. **prompt 版本化 / 回归门禁**：prompts lab 对齐 skill 的 candidate→active promote 门禁，变体即版本。
5. **最小权限 subagent / 上下文隔离**：复用 `SubAgentTemplate.toolIds` 白名单 + `dataScope:"clean_data"` 红线，评测=验证「最小权限下仍达标」。
6. **Observability：token-cost 一等指标**：四 lab `*Summary` 统一带 `avgTotalTokens`/`avgTotalCost`/`avgDurationSec`（skill 已有，对齐全部）。
7. **Offline 套件 → CI 回归门禁**：四 lab 全实装 case集/variant集 CRUD + 历史 + 归档，为后续接 CI gate 留接口。
8. **对抗 / red-team（P3 增量·留位不实现）**：prompt 注入鲁棒性、护栏绕过——hooks/prompts lab 留 expectation 扩展位。

> **取舍**：四 lab「完整测评」= skill/tool 现有闭环 **+ 趋势2（subagents 轨迹/预算断言）+ 趋势3（hooks verdict 断言）** 两项新增 expectation。趋势 1/4/6/7 由现有骨架直接继承；趋势 5 靠复用模板约束；趋势 8 留位 P3。

### A.1 类型契约（`server/src/types.ts` 追加；`web/src/types.ts` 同步镜像）

```ts
// ---- Prompt evaluations (实验场·prompts lab) ----
export interface PromptVariant {
  id: string;
  label: string;
  promptBody: string;                 // 待测 prompt 文本（取自 prompt_templates 或手填）
  role: "system" | "prefix";          // 注入方式：系统指令 / 用户消息前缀
  templateId?: string;                // 来源模板 id（留痕，可空）
}
export interface PromptEvalTask {     // 形状同 SkillEvalTask
  id: string;
  prompt: string;
  expectedPoints?: string[];
  rubric?: string;
}
export interface PromptEvalSet {
  id: string; workspaceId: string; name: string;
  tasks: PromptEvalTask[]; createdAt: number; updatedAt: number;
}
export type PromptPairwiseVerdict = "win" | "tie" | "loss" | "not_judged";
export interface PromptPairwiseResult {   // 逐字对齐 SkillPairwiseResult
  baselineResultId: string; variantResultId: string; taskId: string; attempt: number;
  verdict: PromptPairwiseVerdict; scoreDelta: number | null;
  baselineScore: number | null; variantScore: number | null;
  confidence: number | null; reason: string;
  error: EvaluationError | null; judgeRuns?: PromptPairwiseResult[];
}
export interface PromptPairwiseSummary {  // 逐字对齐 SkillPairwiseSummary
  variantId: string; variantLabel: string; judged: number; skipped: number;
  win: number; tie: number; loss: number; avgScoreDelta: number; avgConfidence: number | null;
}
export interface PromptEvaluationRunResult {  // = SkillEvaluationRunResult 去 skillPaths/activation
  id: string; variantId: string; variantLabel: string; taskId: string; attempt: number;
  status: "success" | "failed"; startedAt: number; endedAt: number; durationSec: number;
  totalTokens: number; totalCost: number; toolCalls: number; outputChars: number; output: string;
  pairwise: PromptPairwiseResult | null; error: EvaluationError | null;
}
export interface PromptVariantSummary {   // = SkillVariantSummary 去 activationRate
  variantId: string; variantLabel: string; total: number; success: number; failed: number;
  avgDurationSec: number; avgTotalTokens: number; avgTotalCost: number;
  avgToolCalls: number; avgOutputChars: number;
}
export interface PromptTaskSummary {
  taskId: string; total: number; success: number; failed: number;
}
export interface PromptEvaluation {       // 去 contextPrefix
  evaluationId: string; workspaceId: string; model: string; repeat: number;
  status: "success" | "failed"; startedAt: number; endedAt: number; durationSec: number;
  variants: PromptVariant[]; tasks: PromptEvalTask[];
  variantSummaries: PromptVariantSummary[]; taskSummaries: PromptTaskSummary[];
  pairwiseSummaries: PromptPairwiseSummary[];
}
export interface PromptEvaluationDetail extends PromptEvaluation {
  results: PromptEvaluationRunResult[];
}
```
口径说明：`PromptVariant` 替 `SkillVariant`（去 skillPaths/retrieval，换 promptBody+role）；结果/汇总去掉 skill 特有的 `activation`/`activationRate`；评测去掉 `contextPrefix`（变体 promptBody 即前缀）。`EvaluationError` 两侧均已存在，直接引用。

export type HookExpectation =
  | { kind: "must-block"; reasonPattern?: string }                       // 预期 tool_call 被拦截（可选 reason 正则）

  id: string; name: string;
  event: HookEvent;                      // 生命周期事件（types.ts HookEvent）

  expected: HookExpectation;
}
export interface HookEvalSet {
  id: string; workspaceId: string; name: string;
  cases: HookEvalCase[]; createdAt: number; updatedAt: number;
}
// 纯 verdict（D.4 evaluateHookFixture 产出，无副作用执行）
export interface HookVerdict {
  matchedHookIds: string[];
  blocked: boolean; blockReason: string | null;
  mutatedInput: Record<string, unknown> | null;   // 有 mutate 命中时=应用 set 后的 input 副本；否则 null

}
export interface HookEvaluationRunResult extends HookVerdict {
  id: string; caseId: string; caseName: string; attempt: number;
  status: "success" | "failed"; startedAt: number; endedAt: number; durationSec: number;
  expectation: HookExpectation; error: EvaluationError | null;
}
export interface HookCaseSummary {        // = ToolCaseSummary
  caseId: string; caseName: string; total: number; success: number; failed: number; avgDurationSec: number;
}
export interface HookEvaluation {
  evaluationId: string; workspaceId: string; repeat: number;
  status: "success" | "failed"; startedAt: number; endedAt: number; durationSec: number;
  cases: HookEvalCase[]; caseSummaries: HookCaseSummary[];
}
export interface HookEvaluationDetail extends HookEvaluation {
  results: HookEvaluationRunResult[];
}
